// Procura por canvas ou img que contenha QR code const qrElements = document.querySelectorAll( 'canvas[style*="background"], img[alt*="qr"], img[alt*="QR"], [data-qrcode], .qr-code' ); // Também procura por elementos com atributos data que contenham qrcode/pix const allDivs = document.querySelectorAll('div, img, canvas'); let foundQR = false; for (let el of allDivs) { const html = el.outerHTML || ""; const attrs = Array.from(el.attributes || []) .map((a) => a.value) .join(" "); // Se encontrar dados de PIX ou QR code no elemento if ( html.includes("data:image/png") || html.includes("qrcode") || html.includes("pix") || attrs.includes("qr") ) { foundQR = true; break; } } if (foundQR || qrElements.length > 0) { if (alreadyTrackedQR()) { clearInterval(checkQRCode); return; } // ✅ Dispara o evento Purchase quando QR code é gerado const value = 64.73; // <-- coloque o valor que você quer atribuir const currency = "BRL"; if (typeof window.fbq === "function") { // Libera 1 disparo de Purchase por sessão apenas neste momento sessionStorage.setItem(ALLOW_KEY, "1"); window.fbq("track", "Purchase", { value, currency }); markTrackedQR(); console.log("[Meta] Purchase disparado na geração do QR code:", value); } else { console.log("[Meta] fbq não encontrado ao gerar QR code"); } clearInterval(checkQRCode); } } catch (err) { console.log("[Meta] erro ao monitorar QR code:", err); } }, 500); // Verifica a cada 500ms // Para de verificar depois de 30 segundos setTimeout(function () { clearInterval(checkQRCode); }, 30000); } // Inicia o monitoramento quando a página carrega if (document.readyState === "loading") { document.addEventListener("DOMContentLoaded", monitorQRCodeGeneration); } else { monitorQRCodeGeneration(); } })();
